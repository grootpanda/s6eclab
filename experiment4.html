1. UNIT PULSE

import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline
n = np.arange(-5, 6)
x = np.zeros_like(n)
x[n == 0] = 1  # position 1 at the center
plt.stem(n, x)  # stem plot of signal x
plt.xticks(n);


2. DISCRETE PULSE

n = np.arange(-5, 6)
x = np.zeros_like(n)
x[(n >= 0) & (n <= 4)] = 1  # position the tp from 0 to 4
x[(n >= -4) & (n <= -1)] = -1  # position the time pulse from -4 to -1
plt.stem(n, x)
plt.xticks(n);


3. DISCRETE PULSE

n = np.arange(0, 6)
x = np.zeros_like(n)
x[n] = n
plt.stem(n, x)
plt.xticks(n);
